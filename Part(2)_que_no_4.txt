/* Step1: Contacting the AWS through Terraform: */
provider "aws" {
  region = "ap-south-1"
  profile = "userprofile"
}

/* Step2: Creating a VPC: */
resource "aws_vpc" "ownvpc" {
  cidr_block       = "192.168.0.0/16"
  instance_tenancy = "default"
  enable_dns_hostnames = true
}

/* Step3: Creating subnets: */
public subnet:
resource "aws_subnet" "public" {
  vpc_id     = aws_vpc.ownvpc.id
  cidr_block = "192.168.1.0/24"
  availability_zone = "ap-south-1a"
}
private subnet:
resource "aws_subnet" "private" {
    vpc_id = aws_vpc.ownvpc.id
    cidr_block = "192.168.0.0/24"
    availability_zone = "ap-south-1b"
}

/* Step-4: Creating internet gateway: */
resource "aws_internet_gateway" "mygw" {
  vpc_id = aws_vpc.ownvpc.id
}

/* Step-5: Creating Route Table: */
to associate it with a public subnet using internet gateway as a target:
resource "aws_route_table" "my_route_table1" {
  vpc_id = aws_vpc.ownvpc.id


  route {
    cidr_block = "0.0.0.0/0"
    gateway_id = aws_internet_gateway.mygw.id
  }
}

resource "aws_route_table_association" "route_table_association1" {
  subnet_id      = aws_subnet.public.id
  route_table_id = aws_route_table.my_route_table1.id
}
to associate it with a private subnet using nat gateway as a target:
resource "aws_route_table" "my_route_table2" {
  vpc_id = aws_vpc.ownvpc.id
  depends_on = [aws_nat_gateway.mynatgw]


  route {
    cidr_block = "0.0.0.0/0"
    nat_gateway_id = aws_nat_gateway.mynatgw.id
  }
}

/* Step-6: Creating EIP to associate it with NAT Gateway: */
resource "aws_eip" "nat" {
  vpc      = true
  depends_on = [aws_internet_gateway.mygw,]

}

/* Step-7: Creating NAT gateway and allocate it the created EIP address: */
resource "aws_nat_gateway" "mynatgw" {
  allocation_id = aws_eip.nat.id
  subnet_id     = aws_subnet.public.id
  depends_on = [aws_internet_gateway.mygw,]

}

/* Step-8: Creating Security Groups: */
"myweb_sg" having inbound rule HTTP for apache web server , SSH for remote login and All ICMP - IPv4 for pinging:
resource "aws_security_group" "mywebsecurity" {
  name        = "my_web_security"
  description = "Allow http,ssh,icmp"
  vpc_id      = aws_vpc.ownvpc.id

  ingress {
    description = "HTTP"
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
  ingress {
    description = "SSH"
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
  ingress {
    description = "ALL ICMP - IPv4"
    from_port   = -1    
    to_port     = -1
    protocol    = "ICMP"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks =  ["0.0.0.0/0"]
  }
  
  tags = {
    Name = "myweb_sg"
  }
}
"mysqlserver_sg" having inbound rule MYSQL/Aurora for database server so that our wordpress vm that is in public subnet can connect with the same:
resource "aws_security_group" "mysqlsecurity" {
  name        = "my_sql_server_security"
  description = "Allow mysql"
  vpc_id      = aws_vpc.ownvpc.id


  ingress {
    description = "MYSQL/Aurora"
    from_port   = 3306
    to_port     = 3306
    protocol    = "tcp"
    security_groups = ["${aws_security_group.mywebsecurity.id}"]
  }
  
  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks =  ["0.0.0.0/0"]
  }
  
  tags = {
   Name = "mysqlserver_sg"  
 }
}

/* Step-9: Creating a EC2 instance with Wordpess setup having the security group allowing port 80 to connect to wordpress site. */
resource "aws_instance" "wordpress" {
  ami           = "ami-000cbce3e1b899ebd"
  instance_type = "t2.micro"
  associate_public_ip_address = true
  subnet_id = aws_subnet.public.id
  vpc_security_group_ids = ["${aws_security_group.mywebsecurity.id}"]
  key_name = "mynewkey"
  availability_zone = "ap-south-1a"


  tags = {
    Name = "wordpress"
  }


}

/* Step-10: Creating an EC2 instance with MySQL setup configured and attaching the security group allowing port 3306 in private subnet to connect with the Wordpress VM: */
resource "aws_instance" "mysql" {
  ami           = "ami-08706cb5f68222d09"
  instance_type = "t2.micro"
  subnet_id = aws_subnet.private.id
  vpc_security_group_ids = ["${aws_security_group.mysqlsecurity.id}"]
  key_name = "mynewkey"
  availability_zone = "ap-south-1b"
 tags = {
    Name = "mysql"
  }
}
