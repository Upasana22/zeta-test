The 3-tiers will be:
	First tier: This will host the web application.
	Second tier: This will host the api (REST api)
	Third tier: This will host the database.

Docker file to containerize our web app
FROM node:10-alpine
WORKDIR /usr/src/app
COPY package*.json ./
RUN npm install
COPY . .
EXPOSE 3001
CMD [ "node", "index.js" ] 
Docker compose.yml file
version: "3.7"
services:
 	 api:
   		build: ./backend
   	 	environment:
      			- CONNECTION_STRING=postgres://demo_user:demo_user@db/demo_db
   	 	depends_on:
     			 - db
    		networks:
     			 - network-backend
     			 - network-frontend
 	webapp:
  		build: ./frontend
    		environment:
      			- API_URL=http://api:3001/data
   		 depends_on:
    			  - api
   		 ports:
     			 - "3000:3000"
  		  networks:
     			 - network-frontend

  	db:
    		image: postgres:11.2-alpine
    		environment:
      			POSTGRES_USER: demo_user
      			POSTGRES_PASSWORD: demo_user
      			POSTGRES_DB: demo_db
   	 	volumes:
   			 - ./init_sql_scripts/:/docker-entrypoint-initdb.d
    		networks:
      			- network-backend

networks:
  		network-backend:
  		network-frontend:
To run compose.yml file
$ docker-compose up –build \
--replicas=4 \
  --name devtest-service \
  --mount source=file2
